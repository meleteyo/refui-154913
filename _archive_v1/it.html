// SubscriptionApplicationService.java
@Service
@RequiredArgsConstructor
public class SubscriptionApplicationService {
    private final PaymentGatewayPort pg;
    private final SubscriptionRepository subRepo;
    private final UserRepository userRepo;
    private final EmailSenderPort email;

    @Transactional
    public Subscription subscribe(UUID userId, UUID planId, String billingKeyToken) {
        User user = userRepo.findActivated(userId)
            .orElseThrow(() -> new BusinessException(USER_NOT_ACTIVATED));

        Plan plan = planRepo.findActive(planId)
            .orElseThrow(() -> new BusinessException(PLAN_NOT_AVAILABLE));

        BillingKey key = pg.issueBillingKey(billingKeyToken);
        Subscription sub = Subscription.create(user, plan, key);

        // 즉시 1회 결제
        Payment payment = Payment.charge(sub, plan.getPrice());
        pg.charge(key, plan.getPrice(), payment.getId() /* 멱등 키 */);

        sub.activate();
        user.upgrade(sub);
        email.sendReceipt(user, payment);

        return subRepo.save(sub);
    }
}

// BillingScheduler.java — C3 컬레보레이션
@Component
@RequiredArgsConstructor
public class BillingScheduler {
    @Scheduled(cron = "0 0 * * * *")
    @SchedulerLock(name = "billing-tick", lockAtMostFor = "30m")
    public void tick() {
        var dueSubs = subRepo.findActiveDueAt(Instant.now());
        for (var sub : dueSubs) {
            try {
                var result = sub.renew(pg);
                if (result.isFailure()) handleFailure(sub);
            } catch (PgTimeoutException e) {
                reconcileQueue.enqueue(sub.getId()); // 멱등성 보장
            }
        }
    }
}
